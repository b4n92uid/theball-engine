00280     // In Terms Of 1/64ths Of Pixels. Thus, To Make A Font

00282     // (h << 6 Is Just A Prettier Way Of Writing h*64)
00283     FT_Set_Char_Size(face, size << 6, size << 6, 96, 96);
00284 
00285     // Here We Ask OpenGL To Allocate Resources For
00286     // All The Textures And Display Lists Which We
00287     // Are About To Create.

00320         //m_charsWidth[i] = width;

00325         // Here We Fill In The Data For The Expanded Bitmap.
00326         // Notice That We Are Using A Two Channel Bitmap (One For
00327         // Channel Luminosity And One For Alpha), But We Assign
00328         // Both Luminosity And Alpha To The Value That We
00329         // Find In The FreeType Bitmap.
00330         // We Use The ?: Operator To Say That Value Which We Use

00332         // Is The FreeType Bitmap Otherwise.

00336             {
00337                 expanded_data[2 * (i + j * width)] = 255;
00338                 expanded_data[2 * (i + j * width) + 1] =
00339                         (i >= bitmap.width || j >= bitmap.rows) ? 0 : bitmap.buffer[i + bitmap.width * j];

00341         }

00344         glBindTexture(GL_TEXTURE_2D, textures[i]);

00350         // We Are Using 2 Channel Data.
00351         glTexImage2D(GL_TEXTURE_2D, 0, 4, width, height, 0, GL_LUMINANCE_ALPHA, GL_UNSIGNED_BYTE, expanded_data);
00352

00354         delete [] expanded_data;

00362         // The Character Has The Right Amount Of Space
00363         // Between It And The One Before It.
00364         glTranslatef(bitmap_glyph->left, 0, 0);
00365 
00366         // Now We Move Down A Little In The Case That The
00367         // Bitmap Extends Past The Bottom Of The Line
00368         // This Is Only True For Characters Like 'g' Or 'y'.
00369         glTranslatef(0, bitmap_glyph->top - bitmap.rows, 0);
00370 
00371         // Now We Need To Account For The Fact That Many Of
00372         // Our Textures Are Filled With Empty Padding Space.
00373         // We Figure What Portion Of The Texture Is Used By

00376         // Quad, We Will Only Reference The Parts Of The Texture
