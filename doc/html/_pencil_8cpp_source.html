00147

00154     // Font Is Called A "face".
00155     FT_Face face;
00156 
00157     // This Is Where We Load In The Font Information From The File.
00158     // Of All The Places Where The Code Might Die, This Is The Most Likely,
00159     // As FT_New_Face Will Fail If The Font File Does Not Exist Or Is Somehow Broken.

00161         throw Exception("Pencil::Load; Open font file error (%s)", path.c_str());

00166     m_charSpace = charsp;
00167     m_fontPath = path;
00168 
00169     // For Some Twisted Reason, FreeType Measures Font Size
00170     // In Terms Of 1/64ths Of Pixels. Thus, To Make A Font
00171     // h Pixels High, We Need To Request A Size Of h*64.
00172     // (h << 6 Is Just A Prettier Way Of Writing h*64)
00173     FT_Set_Char_Size(face, size << 6, size << 6, 96, 96);
00174 
00175     // Here We Ask OpenGL To Allocate Resources For
00176     // All The Textures And Display Lists Which We
00177     // Are About To Create.
00178     m_charList = glGenLists(charindex.size());
00179     GLuint textures[charindex.size()];
00180     glGenTextures(charindex.size(), textures);
00181 
00182     // This Is Where We Actually Create Each Of The Fonts Display Lists.
00183     for(unsigned i = 0; i < charindex.size(); i++)
00184     {
00185         // The First Thing We Do Is Get FreeType To Render Our Character
00186         // Into A Bitmap. This Actually Requires A Couple Of FreeType Commands:
00187         // Load The Glyph For Our Character.
00188 
00189         if(FT_Load_Glyph(face, FT_Get_Char_Index(face, charindex[i]), FT_LOAD_DEFAULT))
00190             throw Exception("Pencil::Load; FT_Load_Glyph failed");
00191 
00192         // Move The Face's Glyph Into A Glyph Node.
00193         FT_Glyph glyph;
00194         if(FT_Get_Glyph(face->glyph, &glyph))
00195             throw Exception("Pencil::Load; FT_Get_Glyph failed");
00196 
00197         // Convert The Glyph To A Bitmap.
00198         FT_Glyph_To_Bitmap(&glyph, ft_render_mode_normal, 0, 1);
00199         FT_BitmapGlyph bitmap_glyph = (FT_BitmapGlyph)glyph;
00200 
00201         // This Reference Will Make Accèssing The Bitmap Easier.
00202         FT_Bitmap & bitmap = bitmap_glyph->bitmap;
00203 
00204         // Use Our Helper Function To Get The Widths Of
00205         // The Bitmap Data That We Will Need In Order To Create
00206         // Our Texture.
00207         int width = math::nextPow2(bitmap.width);
00208         int height = math::nextPow2(bitmap.rows);
00209 
00210         //m_charsWidth[i] = width;
00211 
00212         // Allocate Memory For The Texture Data.
00213         GLubyte * expanded_data = new GLubyte[ 2 * width * height];
00214 
00215         // Here We Fill In The Data For The Expanded Bitmap.
00216         // Notice That We Are Using A Two Channel Bitmap (One For
00217         // Channel Luminosity And One For Alpha), But We Assign
00218         // Both Luminosity And Alpha To The Value That We
00219         // Find In The FreeType Bitmap.
00220         // We Use The ?: Operator To Say That Value Which We Use
00221         // Will Be 0 If We Are In The Padding Zone, And Whatever
00222         // Is The FreeType Bitmap Otherwise.
00223         for(int j = 0; j < height; j++)
00224         {
00225             for(int i = 0; i < width; i++)
00226             {
00227                 expanded_data[2 * (i + j * width)] = 255;
00228                 expanded_data[2 * (i + j * width) + 1] =
00229                         (i >= bitmap.width || j >= bitmap.rows) ? 0 : bitmap.buffer[i + bitmap.width * j];
00230             }
00231         }
00232 
00233         // Now We Just Setup Some Texture Parameters.
00234         glBindTexture(GL_TEXTURE_2D, textures[i]);
00235         glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MAG_FILTER, GL_LINEAR);
00236         glTexParameteri(GL_TEXTURE_2D, GL_TEXTURE_MIN_FILTER, GL_LINEAR);
00237 
00238         // Here We Actually Create The Texture Itself, Notice
00239         // That We Are Using GL_LUMINANCE_ALPHA To Indicate That
00240         // We Are Using 2 Channel Data.
00241         glTexImage2D(GL_TEXTURE_2D, 0, 4, width, height, 0, GL_LUMINANCE_ALPHA, GL_UNSIGNED_BYTE, expanded_data);
00242 
00243         // With The Texture Created, We Don't Need The Expanded Data Anymore.
00244         delete [] expanded_data;
00245 
00246         // Now We Create The Display List
00247         glNewList(m_charList + i, GL_COMPILE);
00248         glBindTexture(GL_TEXTURE_2D, textures[i]);
00249         glPushMatrix();
00250 
00251         // First We Need To Move Over A Little So That
00252         // The Character Has The Right Amount Of Space
00253         // Between It And The One Before It.
00254         glTranslatef(bitmap_glyph->left, 0, 0);
00255 
00256         // Now We Move Down A Little In The Case That The
00257         // Bitmap Extends Past The Bottom Of The Line
00258         // This Is Only True For Characters Like 'g' Or 'y'.
00259         glTranslatef(0, bitmap_glyph->top - bitmap.rows, 0);
00260 
00261         // Now We Need To Account For The Fact That Many Of
00262         // Our Textures Are Filled With Empty Padding Space.
00263         // We Figure What Portion Of The Texture Is Used By
00264         // The Actual Character And Store That Information In
00265         // The x And y Variables, Then When We Draw The
00266         // Quad, We Will Only Reference The Parts Of The Texture
00267         // That Contains The Character Itself.
00268         float x = (float)bitmap.width / (float)width,
00269                 y = (float)bitmap.rows / (float)height;
00270 
00271         // Here We Draw The Texturemapped Quads.
00272         // The Bitmap That We Got From FreeType Was Not
00273         // Oriented Quite Like We Would Like It To Be,
00274         // But We Link The Texture To The Quad
00275         // In Such A Way That The Result Will Be Properly Aligned.
00276         glBegin(GL_QUADS);
00277         glTexCoord2d(0, 0);
00278         glVertex2f(0, bitmap.rows);
00279         glTexCoord2d(0, y);
00280         glVertex2f(0, 0);
00281         glTexCoord2d(x, y);
00282         glVertex2f(bitmap.width, 0);
00283         glTexCoord2d(x, 0);
00284         glVertex2f(bitmap.width, bitmap.rows);
00285         glEnd();
00286 
00287         glPopMatrix();
00288 
00289         glTranslatef((face->glyph->advance.x >> 6) + m_charSpace, 0, 0);
00290 
00291         m_charsWidth[i] = (face->glyph->advance.x >> 6) + m_charSpace;
00292 
00293         // Increment The Raster Position As If We Were A Bitmap Font.
00294         // (Only Needed If You Want To Calculate Text Length)
00295         // glBitmap(0, 0, 0, 0, face->glyph->advance.x >> 6, 0, NULL);
00296 
00297         // Finish The Display List
00298         glEndList();
00299     }
00300 
00301     // We Don't Need The Face Information Now That The Display
00302     // Lists Have Been Created, So We Free The Assosiated Resources.
00303     FT_Done_Face(face);
00304 
00305     manager.instances[this] = path;
