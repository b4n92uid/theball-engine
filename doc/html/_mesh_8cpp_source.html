00571 Vector3f RayCastTriangle(Vector3f p, Vector3f d, Vector3f v0, Vector3f v1, Vector3f v2)
00572 {
00573     Vector3f e1, e2, h, s, q;
00574 
00575     float a, f, u, v;
00576 
00577     e1 = v1 - v0;
00578     e2 = v2 - v0;
00579 
00580     h = Vector3f::cross(d, e2);
00581 
00582     a = Vector3f::dot(e1, h);
00583 
00584     if(a > -0.00001 && a < 0.00001)
00585         return p;
00586 
00587     f = 1.0f / a;
00588     s = p - v0;
00589     u = f * (Vector3f::dot(s, h));
00590 
00591     if(u < 0.0 || u > 1.0)
00592         return p;
00593 
00594     q = Vector3f::cross(s, e1);
00595 
00596     v = f * Vector3f::dot(d, q);
00597 
00598     if(v < 0.0 || u + v > 1.0)
00599         return p;
00600 
00601     // at this stage we can compute t to find out where
00602     // the intersection point is on the line
00603     float t = f * Vector3f::dot(e2, q);
00604 
00605     if(t > 0.00001) // ray intersection
00606         return p + Vector3f::normalize(d) * t;
00607 
00608         // this means that there is a line intersection
00609         // but not a ray intersection
00610     else
00611         return p;
