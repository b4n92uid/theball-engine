00493 Vector3f RayCastTriangle(Vector3f p, Vector3f d, Vector3f v0, Vector3f v1, Vector3f v2)
00494 {
00495     Vector3f e1, e2, h, s, q;
00496 
00497     float a, f, u, v;
00498 
00499     e1 = v1 - v0;
00500     e2 = v2 - v0;
00501 
00502     h = Vector3f::Cross(d, e2);
00503 
00504     a = Vector3f::Dot(e1, h);
00505 
00506     if(a > -0.00001 && a < 0.00001)
00507         return p;
00508 
00509     f = 1.0f / a;
00510     s = p - v0;
00511     u = f * (Vector3f::Dot(s, h));
00512 
00513     if(u < 0.0 || u > 1.0)
00514         return p;
00515 
00516     q = Vector3f::Cross(s, e1);
00517 
00518     v = f * Vector3f::Dot(d, q);
00519 
00520     if(v < 0.0 || u + v > 1.0)
00521         return p;
00522 
00523     // at this stage we can compute t to find out where
00524     // the intersection point is on the line
00525     float t = f * Vector3f::Dot(e2, q);
00526 
00527     if(t > 0.00001) // ray intersection
00528         return p + Vector3f::Normalize(d) * t;
00529 
00530         // this means that there is a line intersection
00531         // but not a ray intersection
00532     else
00533         return p;
00534 }
